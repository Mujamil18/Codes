PROGRAM 10:
int pairSumSequence(int n)
{
    int sum=0;
    for(int i=0; i<n; i++)
    {
        sum += pairSum(i, i+1);
    }
    return sum;
}
int pairSum(int a, int b)
{
    return a+b;
}
int main()
{
    int n=5;
    printf("%d" , pairSumSequence(n));
    return 0;
}
